React Native ActivityIndicator 组件的使用

ActivityIndicator 用来在应用进行一些耗时操作时显示一个旋转的圆形指示器，有以下属性。

  1. animating 决定显不显示indicator, 默认显示。

  2. color 颜色

  3. size 尺寸的可选值为’small’或者’large’, 默认值为’small’。Android平台支持传入数值作为参数。

下面代码简单展示组件及其属性如何使用。

    'use strict';

    import React, { Component } from 'react';
    import {
        View,
        StyleSheet,
        ActivityIndicator,
        Text,
    } from 'react-native';


    export default class ActivityIndicatorExample extends Component {
        //_timer用于保存setInterval函数返回定时器的ID
        _timer: React.PropTypes.number;

        constructor(props) {
            super(props);
        }

        componentWillMount() {
            //组建渲染前初始化组建状态，默认显示Indicator及其下面文字。
            this.setState({
                animating: true,
                opacity: 1
            });
        }

        render() {
            return(
                <View style={styles.container}>
                    <ActivityIndicator
                        animating={this.state.animating}
                        color='mediumspringgreen'
                        size={200}
                    />
                    <View style={{opacity: this.state.opacity}}>
                        <Text>loading</Text>
                    </View>
                </View>
            );
        }

        componentDidMount() {
            //组建加载完成后设定定时函数
            this.ToggleIndicator();
        }

        ToggleIndicator() {
            //每隔两秒切换Indicator显示的状态
            this._timer = setInterval(() => {
                this.setState({
                    animating: !this.state.animating,
                    opacity: this.state.opacity === 1 ? 0 : 1,
                });
            }, 2000);
        }

        componentWillUnmount() {
            //组件卸载前取消由setInterval()方法设置的定时器
            clearInterval(this._timer);
        }
    }

    const styles = StyleSheet.create({
        container: {
            flex: 1,
            justifyContent: 'center',
            alignItems: 'center',
        },
    });
